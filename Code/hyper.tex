\paragraph{Def:} The hyperbolic sine and cosine functions are 
defined by the equations:
\begin{align}
    &\sinh(x) = \frac{e^x-e^{-x}}{2} \\
    &\cosh(x)=\frac{e^x+e^{-x}}{2} \\
    &\tanh(x) = \frac{e^x-e^{-x}}{e^x+e^{-x}}=\frac{\sinh(x)}{\cosh(x)}
\end{align}

\begin{align}
    &\coth(x) = \frac{1}{\tanh(x)}=\frac{\cosh(x)}{\sinh(x)}=\frac{e^x+e^{-x}}{e^x-e^{-x}} \\
    &\sech(x) = \frac{1}{\cosh(x)}=\frac{2}{e^x+e^{-x}} \\
    &\csch(x) = \frac{1}{\sinh(x)}=\frac{2}{e^x-e^{-x}}
\end{align}

\begin{enumerate}
    \item $\cosh^2(x)-\sinh^2(x)=1$
    \item $\cosh^2+\sin^2(x)=\cosh(2x)$
\end{enumerate}

Proof for identity 2:
\begin{align}
    \cosh(x)=\frac{e^x+e^{-x}}{2},\, \sinh(x)&=\frac{e^x-e^{-x}}{2} \\
    \cosh^2(x)+\sinh^2(x)&=(\frac{e^x+e^{-x}}{2})+(\frac{e^x-e^{-x}}{2}) \\
    &= \frac{e^{2x}+2e^{x}e^{-x}+e^{-2x}}{4}+\frac{e^{2x}-2e^{x}e^{-x}+e^{-2x}}{4} \\
    &= \frac{2e^{2x}+2e^{-2x}}{4} \\
    &= \frac{e^{2x}+e^{-2x}}{2}=\cosh(2x)
\end{align}
